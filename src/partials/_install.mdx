import Tabs from '@theme/Tabs';
import TabItem from '@theme/TabItem';

<Tabs groupId="operating-systems">
  <TabItem value="mac" label="macOS">
    On macOS, you can use [Homebrew](https://brew.sh/) to install and update Glasskube.
    ```shell
    brew install glasskube/tap/glasskube
    ```
  </TabItem>
  <TabItem value="linux" label="Linux">
    You can install Glasskube using one of the package managers below.
    <details>
      <summary>RPM-based installation (RedHat/CentOS/Fedora)</summary>

      {/* x-release-please-start-version */}
      ```
      dnf install https://releases.dl.glasskube.dev/glasskube_v0.26.1_amd64.rpm
      ```
    </details>

    <details>
      <summary>DEB-based installation (Ubuntu/Debian)</summary>

      ```
      curl -LO https://releases.dl.glasskube.dev/glasskube_v0.26.1_amd64.deb
      sudo dpkg -i glasskube_v0.26.1_amd64.deb
      ```
    </details>

    <details>
      <summary>APK-based installation (Alpine)</summary>
      ```
      curl -LO https://releases.dl.glasskube.dev/glasskube_v0.26.1_amd64.apk
      apk add --allow-untrusted glasskube_v0.26.1_amd64.apk
      ```
    </details>

    If you are using a distribution that does not use one of the package managers above, or require a 32-bit binary,
    check out additional download options attached to our [latest release](https://github.com/glasskube/glasskube/releases/latest).
  </TabItem>

  <TabItem value="win" label="Windows">
    Download the [windows archive](https://releases.dl.glasskube.dev/glasskube_v0.26.1_windows_x86_64.zip) from our
    latest [Release](https://github.com/glasskube/glasskube/releases/latest) and unpack it using Windows Explorer.
  </TabItem>

  <TabItem value="nixos" label="NixOS/Nixpkgs">
    You can either use the package temporarily in a <b>nix-shell</b>:
    ```
    nix-shell -p glasskube
    ```
    Or install it globally by adding `pkgs.glasskube` to your `environment.systemPackages`.
  </TabItem>
</Tabs>

<br/>

After installing Glasskube on your local machine, make sure to install the necessary components in your Kubernetes cluster by running `glasskube bootstrap`.
For more information, check out our [bootstrap guide](/products/package-manager/docs/getting-started/bootstrap).
